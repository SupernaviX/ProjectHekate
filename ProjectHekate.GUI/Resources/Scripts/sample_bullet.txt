action SomeRandomShit(someShit)
{
    $Angle += PI_180*someShit;
    wait(0);
}

action UpdateBullet()
{
    var angle = 0;
    for(var i = 0; i < 30; i++) {
        angle = i*2*PI/180;
        var sprite = 0;
        sprite = i % 2 == 0 ? 2 : 3;

        fire bullet($X, $Y, $Angle + angle, $Speed, sprite) with updater SomeRandomShit(i);
        wait(0);
    }

    wait(600);
}
